-- ========================================
-- SISTEMA DE CONFIGURAÇÃO (AUTO SAVE/LOAD)
-- ========================================
local ConfigSystem = {
	FileName = "configssl.json",
	Data = {}
}

do
	function ConfigSystem:Save()
		local success, json = pcall(function()
			return Services.HttpService:JSONEncode(self.Data)
		end)
		if success then
			if writefile then
				writefile(self.FileName, json)
			end
		end
	end

	function ConfigSystem:Load()
		if isfile and isfile(self.FileName) then
			local content = readfile(self.FileName)
			local success, decoded = pcall(function()
				return Services.HttpService:JSONDecode(content)
			end)
			if success and type(decoded) == "table" then
				self.Data = decoded
				task.spawn(function()
					repeat task.wait() until _G.Notify
					_G.Notify("Configurações carregadas com sucesso!")
				end)
			end
		end
	end

	ConfigSystem:Load()
end

function ThemeSystem:Register(instance, property, colorKey)
	if not instance then return end
	table.insert(self.Registry, {
		Instance = instance,
		Property = property,
		ColorKey = colorKey
	})
	local theme = self.Themes[self.CurrentThemeName]
	if theme and theme[colorKey] then
		pcall(function()
			instance[property] = theme[colorKey]
		end)
	end
end

function ThemeSystem:Unregister(instance)
	for i = #self.Registry, 1, -1 do
		if self.Registry[i].Instance == instance then
			table.remove(self.Registry, i)
		end
	end
end

function ThemeSystem:OnThemeChanged(callback)
	table.insert(self.ThemeChangedCallbacks, callback)
end

function ThemeSystem:SetTheme(themeName)
	if not self.Themes[themeName] then return end
	self.CurrentThemeName = themeName
	local theme = self.Themes[themeName]
	
	for i = #self.Registry, 1, -1 do
		local entry = self.Registry[i]
		if entry.Instance and entry.Instance.Parent then
			pcall(function()
				entry.Instance[entry.Property] = theme[entry.ColorKey]
			end)
		else
			table.remove(self.Registry, i)
		end
	end
	
	for _, callback in ipairs(self.ThemeChangedCallbacks) do
		pcall(callback, theme)
	end
end

local function CreateTab(tabName)
	if ST_STATE.Tabs[tabName] then return ST_STATE.Tabs[tabName] end
	local tabCount = 0
	for _ in pairs(ST_STATE.Tabs) do tabCount += 1 end
	local btn = Instance.new("TextButton")
	btn.Size = UDim2.fromOffset(0, ST_CFG.TabHeight)
	btn.AutomaticSize = Enum.AutomaticSize.X
	btn.Text = tabName
	btn.TextSize = 12
	btn.Font = Enum.Font.GothamBold
	btn.BorderSizePixel = 0
	btn.LayoutOrder = tabCount + 1
	btn.ZIndex = 11
	btn.Parent = ST_UI.tabBar
	Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 4)
	local btnPad = Instance.new("UIPadding")
	btnPad.PaddingLeft = UDim.new(0, 12)
	btnPad.PaddingRight = UDim.new(0, 12)
	btnPad.Parent = btn
	local container = Instance.new("Frame")
	container.Size = UDim2.new(1, 0, 0, 0)
	container.AutomaticSize = Enum.AutomaticSize.Y
	container.BackgroundTransparency = 1
	container.Visible = false
	container.ZIndex = 11
	container.LayoutOrder = tabCount + 1
	container.Parent = ST_UI.contentArea
	local containerLayout = Instance.new("UIListLayout")
	containerLayout.Padding = UDim.new(0, 10)
	containerLayout.SortOrder = Enum.SortOrder.LayoutOrder
	containerLayout.Parent = container
	local tabData = { btn = btn, container = container, itemCount = 0 }
	ST_STATE.Tabs[tabName] = tabData

	local function updateTabVisuals()
		local theme = ThemeSystem.Themes[ThemeSystem.CurrentThemeName]
		for name, data in pairs(ST_STATE.Tabs) do
			local isActive = name == ST_STATE.ActiveTab
			Services.TweenService:Create(data.btn, TweenInfo.new(0.15, Enum.EasingStyle.Quart), {
				BackgroundColor3 = isActive and theme.Text or theme.Foreground,
				TextColor3 = isActive and theme.Background or theme.MutedText
			}):Play()
		end
	end
	ThemeSystem:OnThemeChanged(updateTabVisuals)

	btn.MouseButton1Click:Connect(function()
		ST_STATE.ActiveTab = tabName
		for name, data in pairs(ST_STATE.Tabs) do
			data.container.Visible = name == tabName
		end
		updateTabVisuals()
	end)
	if ST_STATE.ActiveTab == nil then
		ST_STATE.ActiveTab = tabName
		container.Visible = true
		task.defer(updateTabVisuals)
	end
	return tabData
end

	local function updateTrackLayout()
		local rowW = row.AbsoluteSize.X
		if not rowW or rowW == 0 then return end
		local candidate = math.max(28, math.floor(rowW * 0.30))
		local newW = math.min(TRACK_W, candidate)
		track.Size = UDim2.fromOffset(newW, TRACK_H)
		local th = track.AbsoluteSize.Y > 0 and track.AbsoluteSize.Y or TRACK_H
		local kh = knob.AbsoluteSize.Y > 0 and knob.AbsoluteSize.Y or KNOB_H
		knob.Position = UDim2.fromOffset(PAD, (th - kh) / 2)
		shadow.Position = UDim2.fromOffset(PAD + 1, (th - kh) / 2 + 1)
	end

	row:GetPropertyChangedSignal("AbsoluteSize"):Connect(updateTrackLayout)
	task.defer(updateTrackLayout)

	local function applyVisual(on)
		local theme = ThemeSystem.Themes[ThemeSystem.CurrentThemeName]
		local tw = track.AbsoluteSize.X
		if not tw or tw == 0 then tw = TRACK_W end
		local th = track.AbsoluteSize.Y > 0 and track.AbsoluteSize.Y or TRACK_H
		local kw = knob.AbsoluteSize.X > 0 and knob.AbsoluteSize.X or KNOB_W
		local kh = knob.AbsoluteSize.Y > 0 and knob.AbsoluteSize.Y or KNOB_H
		local onX = math.max(PAD, tw - kw - PAD)
		local offX = PAD
		local onPos = UDim2.fromOffset(onX, (th - kh) / 2)
		local offPos = UDim2.fromOffset(offX, (th - kh) / 2)
		local shadowOn = UDim2.fromOffset(onX + 1, (th - kh) / 2 + 1)
		local shadowOff = UDim2.fromOffset(offX + 1, (th - kh) / 2 + 1)

		Services.TweenService:Create(track, TweenInfo.new(0.18, Enum.EasingStyle.Quart), { BackgroundColor3 = on and theme.Accent or theme.SecondaryBg }):Play()
		Services.TweenService:Create(knob, TweenInfo.new(0.18, Enum.EasingStyle.Quart), { Position = on and onPos or offPos, BackgroundColor3 = on and theme.Background or theme.Text }):Play()
		Services.TweenService:Create(shadow, TweenInfo.new(0.18, Enum.EasingStyle.Quart), { Position = on and shadowOn or shadowOff }):Play()

		trackStroke.Color = on and theme.Border or theme.MutedText
		knobStroke.Color = on and theme.Accent or theme.MutedText
	end

	ThemeSystem:OnThemeChanged(function()
		applyVisual(value)
	end)

	btn.MouseButton1Click:Connect(function()
		value = not value
		applyVisual(value)
		ConfigSystem.Data[configKey] = value
		ConfigSystem:Save()
		if callback then task.spawn(callback, value) end
	end)

	local Toggle = {}
	function Toggle:SetValue(newValue)
		value = newValue
		applyVisual(value)
		ConfigSystem.Data[configKey] = value
		ConfigSystem:Save()
		if callback then task.spawn(callback, value) end
	end
	function Toggle:GetValue() return value end

	task.defer(function()
		if ConfigSystem.Data[configKey] ~= nil then
			Toggle:SetValue(value)
		else
			if value then
				applyVisual(true)
				if callback then task.spawn(callback, true) end
			end
		end
	end)

	return Toggle
end

-- HELPER FUNCTIONS
local function SafeGetChannel(channelIndex)
	if not synchronizerReady then return nil end
	local ok, result = pcall(function()
		-- Primeiro tentar GetChannel direto
		if SynchronizerModule.GetChannel then
			local ch = SynchronizerModule:GetChannel(channelIndex)
			if ch then return ch end
		end
		
		-- Se não, tentar via GetAllChannels ou Create
		local allChannels = SynchronizerModule:GetAllChannels()
		if allChannels and allChannels[channelIndex] then 
			return allChannels[channelIndex] 
		end
		if SynchronizerModule.Create then
			return SynchronizerModule:Create(nil, channelIndex, nil)
		end
	end)
	if ok then return result end
	return nil
end

local function SafeWaitChannel(channelIndex)
	-- Esperar o Synchronizer estar pronto
	local maxSyncWait = 15
	local syncStart = tick()
	while not synchronizerReady and tick() - syncStart < maxSyncWait do
		task.wait(0.2)
	end
	
	if not synchronizerReady then
		warn("[SYNCHRONIZER] Não está pronto após espera!")
		return nil
	end
	
	-- Tentar obter o canal
	local channel = SafeGetChannel(channelIndex)
	if channel then return channel end
	
	-- Esperar o canal aparecer
	local start = tick()
	local maxWait = 10
	repeat
		task.wait(0.2)
		channel = SafeGetChannel(channelIndex)
		if tick() - start > maxWait then 
			warn("[SYNCHRONIZER] Timeout ao esperar canal: " .. tostring(channelIndex))
			return nil 
		end
	until channel
	
	return channel
end

-- Função para formatar números
local function FormatNumber(num)
	if type(num) == "string" then return num end
	if num >= 1e12 then
		return string.format("%.1fT", num / 1e12)
	elseif num >= 1e9 then
		return string.format("%.1fB", num / 1e9)
	elseif num >= 1e6 then
		return string.format("%.1fM", num / 1e6)
	elseif num >= 1e3 then
		return string.format("%.1fK", num / 1e3)
	else
		return tostring(math.floor(num))
	end
end

-- Função para coletar todos os pets
local function GetAllPetsRaw()
	-- Esperar o Synchronizer estar pronto
	if not synchronizerReady then
		local maxWait = 15
		local start = tick()
		while not synchronizerReady and tick() - start < maxWait do
			task.wait(0.2)
		end
		if not synchronizerReady then
			warn("[GET PETS] Synchronizer não está pronto!")
			return {}
		end
	end
	
	local syncFunction = ReplicatedStorage:FindFirstChild("BestPetsSynchronizerFunction")
	if syncFunction and syncFunction:IsA("BindableFunction") then
		local ok, result = pcall(function()
			return syncFunction:Invoke({Mode = "All"})
		end)
		if ok and result and result.Pets and result.Pets.All then
			local all = {}
			for _, pet in ipairs(result.Pets.All) do
				table.insert(all, {
					name = pet.Name,
					rarity = pet.Rarity,
					mutation = pet.Mutation or "",
					traits = pet.Traits or {},
					genVal = pet.Generation or 0,
					index = pet.Index,
					plotName = pet.Plot,
				})
			end
			return all
		end
	end
	
	local all = {}
	if not Plots then return all end
	for _, plot in Plots:GetChildren() do
		if not (plot:IsA("Model") and plot:GetAttribute("Tier")) then continue end
		local channel = SafeWaitChannel(plot.Name)
		if not channel then 
			warn("[GET PETS] Canal não encontrado para plot: " .. plot.Name)
			continue 
		end
		
		local animalList = channel:Get("AnimalList") or {}
		for _, entry in animalList do
			if type(entry) ~= "table" or not entry.Index or entry == "Empty" then continue end
			local data = AnimalsData[entry.Index]
			local name = (data and data.DisplayName) or entry.Index
			local rarity = (data and data.Rarity) or "Common"
			local genVal = GetGeneration(entry.Index, entry.Mutation, entry.Traits)
			table.insert(all, {
				name = name,
				rarity = rarity,
				mutation = entry.Mutation or "",
				traits = entry.Traits or {},
				genVal = genVal,
				index = entry.Index,
				plotName = plot.Name,
			})
		end
	end
	table.sort(all, function(a, b) return a.genVal > b.genVal end)
	return all
end

local function GetAnimalIndexFromPayload(plot, petIndex)
	if not plot or not petIndex then return nil end
	local podiums = plot:FindFirstChild("AnimalPodiums")
	if not podiums then return nil end
	local channel = SafeWaitChannel(plot.Name)
	if not channel then return nil end
	local animalList = channel:Get("AnimalList") or {}
	for slot, entry in pairs(animalList) do
		if type(entry) == "table" and entry.Index == petIndex then
			local slotNum = tonumber(slot)
			if slotNum then return slotNum end
		end
	end
	return nil
end

end

	local bestPet = allPets[1]
	local plotName = bestPet.plotName
	local petIndex = bestPet.index
	if not plotName or not Plots then return nil, nil end
	local plot = Plots:FindFirstChild(plotName)
	if not plot then return nil, nil end
	
	-- Esperar o canal estar disponível
	local channel = SafeWaitChannel(plot.Name)
	if not channel then return nil, nil end
	
	local index = GetAnimalIndexFromPayload(plot, petIndex)
	if not index then index = GetAnimalIndex(plot) end
	return plot, index
end

local function GetPriorityPlotAndIndex()
	if not synchronizerReady or #priorityList == 0 then return nil, nil end
	local allPets = GetAllPetsRaw()
	for _, petIndex in ipairs(priorityList) do
		for _, pet in ipairs(allPets) do
			if pet.index == petIndex then
				local plot = Plots:FindFirstChild(pet.plotName)
				if plot then
					local index = GetAnimalIndexFromPayload(plot, pet.index)
					if not index then index = GetAnimalIndex(plot) end
					return plot, index
				end
			end
		end
	end
	return nil, nil
end

local function GetNearestPlotAndIndex()
	if not synchronizerReady then return nil, nil end
	local allPets = GetAllPetsRaw()
	if #allPets == 0 then return nil, nil end
	
	local lp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
	if not lp then return GetHighestPlotAndIndex() end
	
	local nearestPet = nil
	local minDist = math.huge
	
	for _, pet in ipairs(allPets) do
		local plot = Plots:FindFirstChild(pet.plotName)
		if plot then
			local dist = (lp.Position - plot:GetPivot().Position).Magnitude
			if dist < minDist then
				minDist = dist
				nearestPet = pet
			end
		end
	end
	
	if nearestPet then
		local plot = Plots:FindFirstChild(nearestPet.plotName)
		local index = GetAnimalIndexFromPayload(plot, nearestPet.index)
		if not index then index = GetAnimalIndex(plot) end
		return plot, index
	end
	return nil, nil
end

-- VIEWPORT LOADER
local function LoadAnimalInViewport(vp, petName)
	task.spawn(function()
		pcall(function()
			if not AnimalModels then return end
			local tmpl = AnimalModels:FindFirstChild(petName)
			if not tmpl then return end
			local clone = tmpl:Clone()
			local wm = Instance.new("WorldModel")
			wm.Parent = vp
			clone.Parent = wm
			if clone.PrimaryPart then clone.PrimaryPart.Anchored = true end
			for _, d in ipairs(clone:GetDescendants()) do
				if d:IsA("BasePart") then
					d.Anchored = true; d.CanCollide = false
					d.CastShadow = false; d.Massless = true
				end
			end
			local ok, bbCF, bbSize = pcall(function() return clone:GetBoundingBox() end)
			if not ok then bbCF = clone:GetPivot(); bbSize = Vector3.new(4,4,4) end
			local sz = math.max(bbSize.X, bbSize.Y, bbSize.Z)
			local fov = 50
			local dist = (sz * 0.5) / math.tan(math.rad(fov * 0.5)) * 0.85
			local offset = Vector3.new(1, 0.4, 1).Unit
			local cam = Instance.new("Camera")
			cam.FieldOfView = fov
			cam.CFrame = CFrame.new(bbCF.Position + offset * (dist + sz * 0.5), bbCF.Position)
			cam.Parent = vp
			vp.CurrentCamera = cam
			if AnimalAnimations then
				local animFolder = AnimalAnimations:FindFirstChild(petName)
				if animFolder then
					local idleAnim = animFolder:FindFirstChild("Idle") or animFolder:FindFirstChild("idle") or animFolder:GetChildren()[1]
					if idleAnim then
						local animCtrl = clone:FindFirstChildWhichIsA("AnimationController", true)
						if not animCtrl then
							animCtrl = Instance.new("AnimationController")
							animCtrl.Parent = clone
						end
						local animator = animCtrl:FindFirstChildOfClass("Animator")
						if not animator then
							animator = Instance.new("Animator")
							animator.Parent = animCtrl
						end
						local track = animator:LoadAnimation(idleAnim)
						track.Looped = true; track:Play(0)
						if track.Length > 0 then track.TimePosition = os.clock() % track.Length end
						task.spawn(function()
							while vp.Parent do
								task.wait(1)
								pcall(function()
									if track.Length > 0 then
										local t = os.clock() % track.Length
										if math.abs(t - track.TimePosition) > 0.05 then
											track.TimePosition = t
										end
									end
								end)
							end
						end)
					end
				end
			end
		end)
	end)
end

local function findAdorneeGlobal(animalData)
	if not animalData then return nil end
	if not synchronizerReady then 
		warn("[FIND ADORNEE] Synchronizer não está pronto!")
		return nil 
	end
	
	local plot = Plots and Plots:FindFirstChild(animalData.plotName)
	if not plot then return nil end
	
	-- Procurar o model do pet diretamente no plot pelo nome
	for _, child in ipairs(plot:GetChildren()) do
		if child:IsA("Model") and child.Name == animalData.name then
			-- Retorna a PrimaryPart ou primeira BasePart do model
			if child.PrimaryPart then
				return child.PrimaryPart
			end
			local basePart = child:FindFirstChildWhichIsA("BasePart", true)
			if basePart then
				return basePart
			end
		end
	end
	
	warn("[FIND ADORNEE] Model do pet não encontrado no plot:", animalData.name)
	return nil
end

local function getClosestBaseIdx(pos)
	local closest, dist = 1, math.huge
	for i, basePos in pairs(BASES_LOW) do
		local d = (Vector2.new(pos.X, pos.Z) - Vector2.new(basePos.X, basePos.Z)).Magnitude
		if d < dist then dist = d; closest = i end
	end
	return closest
end

local function walkForward(seconds)
	local char = player.Character
	local hum = char and char:FindFirstChild("Humanoid")
	local hrp = char and char:FindFirstChild("HumanoidRootPart")
	if not hum or not hrp then return end

	local lookVector = hrp.CFrame.LookVector
	local distance = (hum.WalkSpeed or 16) * (seconds or 0.4)

	local params = RaycastParams.new()
	params.FilterDescendantsInstances = {char}
	params.FilterType = Enum.RaycastFilterType.Blacklist
	params.IgnoreWater = true

	local origin = hrp.Position
	local result = workspace:Raycast(origin, lookVector * distance, params)

	local targetPos
	if result and result.Position then
		targetPos = result.Position - lookVector * 0.5
	else
		targetPos = origin + lookVector * distance
	end

	hrp.CFrame = CFrame.lookAt(targetPos, targetPos + lookVector)
	hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
	hum:Move(Vector3.zero, false)
end

local function runAutoSnipe(targetPetData)
	if isTpMoving then return end
	if not targetPetData then return end

	local char = player.Character
	local hrp = char and char:FindFirstChild("HumanoidRootPart")
	local hum = char and char:FindFirstChild("Humanoid")
	
	if not hrp or not hum or hum.Health <= 0 then return end
	
	isTpMoving = true
	
	local targetPart = findAdorneeGlobal(targetPetData)
	if not targetPart then 
		isTpMoving = false
		return 
	end
	
	local exactPos = targetPart.Position
	local carpet = player.Backpack:FindFirstChild("Flying Carpet") or char:FindFirstChild("Flying Carpet")

	if carpet then hum:EquipTool(carpet) end
	task.wait(0.01)
	
	local isSecondFloor = exactPos.Y > 10
	local plotIndex = getClosestBaseIdx(exactPos)
	local targetBasePos = isSecondFloor and BASES_HIGH[plotIndex] or BASES_LOW[plotIndex]
	
	local minHeight = TPConfig.MaxHeight
	local targetHeight = math.max(targetBasePos.Y, minHeight)

	-- Impulso para cima
	local jumpStart = tick()
	while hrp.Position.Y < targetHeight and (tick() - jumpStart) < 2 do
		hrp.AssemblyLinearVelocity = Vector3.new(hrp.AssemblyLinearVelocity.X, TPConfig.UpSpeed, hrp.AssemblyLinearVelocity.Z)
		Services.RunService.Heartbeat:Wait()
	end

	-- TP para a base
	for i = 1, 1 do
		hrp.AssemblyLinearVelocity = Vector3.new(hrp.AssemblyLinearVelocity.X, 0, hrp.AssemblyLinearVelocity.Z)
		if (hrp.Position - targetBasePos).Magnitude > 3 then
			hrp.CFrame = CFrame.new(targetBasePos)
			task.wait(TPConfig.DelayFrontBase)
		end
	end

	-- Posição de clone (só primeiro andar)
	if not isSecondFloor then
		local bestSpot = CLONE_POSITIONS_FLOOR[1]
		local minDst = math.huge
		for _, v in ipairs(CLONE_POSITIONS_FLOOR) do
			local d = (targetPart.Position - v).Magnitude
			if d < minDst then minDst = d; bestSpot = v end
		end
		for i = 1, 1 do
			if (hrp.Position - bestSpot).Magnitude > 3 then
				hrp.CFrame = CFrame.new(bestSpot)
				task.wait(TPConfig.DelayFrontBase)

local function tpToBestBrainrot()
	if isTpMoving then return end

	local Players = game:GetService("Players")
	local RunService = game:GetService("RunService")
	local char = player.Character
	local humanoid = char and char:FindFirstChildOfClass("Humanoid")
	local backpack = player:FindFirstChildOfClass("Backpack")

	if not char or not humanoid or not backpack then
		return
	end

	local function getTool(name)
		return char:FindFirstChild(name) or backpack:FindFirstChild(name)
	end

	local bat = getTool("Bat")
	local quantum = getTool("Quantum Cloner")

	if not bat or not quantum then
		return
	end

	pcall(function() sethiddenproperty(player, "MaxSimulationRadius", 100123123000) end)
	pcall(function() sethiddenproperty(player, "SimulationRadius", 10012312300) end)
	setsimulationradius(100123123000, 10012312300)

	local bestPet = GetAllPetsRaw()[1]
	if not bestPet then 
		return 
	end
	
	local plot = Plots and Plots:FindFirstChild(bestPet.plotName)
	if not plot then 
		return 
	end
		
	local targetPart = findAdorneeGlobal(bestPet)
	if not targetPart then 
		return 
	end

	isTpMoving = true

	local cloneSpawned = false
	local cloneConnection

	local function processClone(child)
		if not child or not child:IsA("Model") then return end
		local nameLower = string.lower(child.Name)
		local myNameLower = string.lower(player.Name)
		local myDisplayNameLower = string.lower(player.DisplayName)
		local myUserIdStr = tostring(player.UserId)
		
		local hasClone = string.find(nameLower, "clone")
		local isMine = string.find(nameLower, myNameLower) 
			or string.find(nameLower, myDisplayNameLower) 
			or string.find(nameLower, myUserIdStr)
			
		if not (hasClone and isMine) then return end
		if cloneSpawned then return end
		
		cloneSpawned = true
		if cloneConnection then cloneConnection:Disconnect() end
		
		-- 1. Espera o HumanoidRootPart e o Humanoid carregarem completamente
		local hrp = child:WaitForChild("HumanoidRootPart", 5) or child:FindFirstChildWhichIsA("BasePart")
		local hum = child:WaitForChild("Humanoid", 5) or child:FindFirstChildOfClass("Humanoid")
		
		if hum and hrp then
			
			-- 4. Checa network, se for true, teleporta o clone até o melhor brainrot
			if isnetworkowner(child.PrimaryPart) then
				hrp.CFrame = CFrame.new(targetPart.Position)
				child:PivotTo(CFrame.new(targetPart.Position))
				hum:ChangeState(Enum.HumanoidStateType.Ragdoll)
				
				-- Usa o quantum cloner imediatamente quando detecta proximidade com o pet + networkowner true
				local startCheck = tick()
				local usedCloner = false
				
				while tick() - startCheck < 5 do
					if not child or not child.Parent then break end
					
					-- Mantém o clone no alvo
					hrp.CFrame = CFrame.new(targetPart.Position)
					child:PivotTo(CFrame.new(targetPart.Position))
					hum:ChangeState(Enum.HumanoidStateType.Ragdoll)
					
					local dist = (hrp.Position - targetPart.Position).Magnitude
					if dist < 15 and isnetworkowner(child.PrimaryPart) then
						-- Usa o quantum cloner imediatamente
						quantum.Parent = char
						pcall(function()
							if Network.TeleportRemote then 
								Network.TeleportRemote:FireServer() 
							end
						end)
						usedCloner = true
						
						-- Dispara o ProximityPrompt e o remote de steal
						task.spawn(function()
							local plotIndex = GetAnimalIndexFromPayload(plot, bestPet.index)
							if not plotIndex then plotIndex = GetAnimalIndex(plot) end
							if plotIndex then
								FireStealRemotes(plot, plotIndex, true)
							end
						end)
						break
					end
					task.wait()
				end
			end
		end
		
		-- Finalização e limpeza das ferramentas
		task.wait(0.2)
		quantum.Parent = backpack
		bat.Parent = backpack
		isTpMoving = false
	end

	-- Verifica se o clone já existe no workspace antes de iniciar o loop de swap
	for _, child in ipairs(workspace:GetChildren()) do
		if child:IsA("Model") and child.Name then
			local nameLower = string.lower(child.Name)
			local myNameLower = string.lower(player.Name)
			local myDisplayNameLower = string.lower(player.DisplayName)
			local myUserIdStr = tostring(player.UserId)
			
			local hasClone = string.find(nameLower, "clone")
			local isMine = string.find(nameLower, myNameLower) 
				or string.find(nameLower, myDisplayNameLower) 
				or string.find(nameLower, myUserIdStr)
				
			if hasClone and isMine then
				processClone(child)
				break
			end
		end
	end

	-- Se o clone ainda não foi localizado, inicia escuta por novos clones
	if not cloneSpawned then
		cloneConnection = workspace.ChildAdded:Connect(processClone)
		
		-- Executa a sequência de glitch swap para spawnar o clone de forma garantida
		task.spawn(function()
			while not cloneSpawned and isTpMoving do
				for i = 1, 31 do
					if cloneSpawned or not isTpMoving then break end
					quantum.Parent = backpack
					bat.Parent = char
					bat.Parent = backpack
					quantum.Parent = char					
				end
				task.wait()
				task.delay(1, function() pcall(function() quantum:Activate() end) end)
			end
		end)
	end

	-- Espera até que o clone seja processado ou timeout
	local startWait = tick()
	while not cloneSpawned and isTpMoving and (tick() - startWait < 8) do
		RunService.Heartbeat:Wait()
	end

	-- Limpeza de segurança após 10 segundos caso nada aconteça
	task.delay(10, function()
		if cloneConnection then cloneConnection:Disconnect() end
		isTpMoving = false
		bat.Parent = backpack
		quantum.Parent = backpack
	end)
end

end)

local function normalizeText(value)
	return string.lower(tostring(value or ""))

	local normalized = normalizeText(text)

local function getTripOwnerName(instanceName)
	if type(instanceName) ~= "string" or string.sub(instanceName, 1, #PREFIX) ~= PREFIX then
		return nil
	end
	local suffix = string.sub(instanceName, #PREFIX + 1)
	return suffix ~= "" and suffix or nil
end

	if instance:IsA("BasePart") then return instance end

	for _, conn in ipairs(espConnections) do
		conn:Disconnect()

	espConnections = {}
	
	if not subspaceMineESPEnabled then
		-- Remove todos os billboards
		for target, _ in pairs(billboardCache) do
			removeBillboard(target)
		end
		billboardCache = {}
		return

		if subspaceMineESPEnabled then
			watchInstance(child)
		end

	local removeConn = toolsAdds.ChildRemoved:Connect(function(child)
		removeBillboard(child)

	local normalized = normalizeText(text)

end

local function findSignContainer(plotModel)
	return plotModel:FindFirstChild("PlotSing", true) or plotModel:FindFirstChild("PlotSign", true)
end

local function plotBelongsToPlayer(plotModel)
	if textMatchesPlayer(plotModel.Name) then
		return true
	end
	
	local signContainer = findSignContainer(plotModel)
	if not signContainer then
		return false
	end
	
	for _, descendant in ipairs(signContainer:GetDescendants()) do
		if descendant:IsA("TextLabel") or descendant:IsA("TextButton") or descendant:IsA("TextBox") then
			if textMatchesPlayer(descendant.Text) then
				return true
			end
		elseif descendant:IsA("StringValue") then
			if textMatchesPlayer(descendant.Value) then
				return true
			end
		end
	end
	
	return false
end

local function findTargetPart(plotModel)
	local signContainer = findSignContainer(plotModel)
	if signContainer then
		if signContainer:IsA("BasePart") then
			return signContainer
		end
		return signContainer:FindFirstChildWhichIsA("BasePart", true)
	end
	
	if plotModel.PrimaryPart then
		return plotModel.PrimaryPart
	end
	
	return plotModel:FindFirstChildWhichIsA("BasePart", true)
end

	local plotsFolder = Services.Workspace:FindFirstChild("Plots")

	return nil

local function updateBeam()
	if not myBaseESPEnabled then
		local sourceAttachment = getSourceAttachment()
		if sourceAttachment then
			local existingBeam = sourceAttachment:FindFirstChild(BEAM_NAME)
			if existingBeam then
				existingBeam.Enabled = false
			end
		end
		return
	end
	
	local ownedPlot = findOwnedPlot()
	local sourceAttachment = getSourceAttachment()
	
	if not ownedPlot or not sourceAttachment then
		local existingBeam = sourceAttachment and sourceAttachment:FindFirstChild(BEAM_NAME)
		if existingBeam then existingBeam.Enabled = false end
		return
	end
	
	local targetPart = findTargetPart(ownedPlot)
	if not targetPart then return end
	
	local targetAttachment = getTargetAttachment(targetPart)
	local beam = getBeam(sourceAttachment)
	beam.Attachment1 = targetAttachment
	beam.Enabled = true
end

local function cachePlotParts(plotModel)
	cachedParts = {}
	for _, obj in ipairs(plotModel:GetDescendants()) do
		if obj:IsA("BasePart") and obj.Transparency < 1 then
			table.insert(cachedParts, obj)
		end
	end
end

local function updateGradient()
	if not myBaseESPEnabled then return end
	
	gradientT += 0.05 * gradientDir
	if gradientT >= 1 then
		gradientT = 1
		gradientDir = -1
	elseif gradientT <= 0 then
		gradientT = 0
		gradientDir = 1
	end
	
	local color = COLOR_A:Lerp(COLOR_B, gradientT)
	for _, part in ipairs(cachedParts) do
		if part and part.Parent then
			part.Color = color
		end
	end
end

local function setupMyBaseESP()
	-- Para loops existentes
	if beamUpdateLoop then
		beamUpdateLoop = false
	end
	if gradientUpdateLoop then
		gradientUpdateLoop = false
	end
	
	if not myBaseESPEnabled then
		-- Desabilita beam
		updateBeam()
		cachedParts = {}
		return
	end
	
	-- Atualiza beam imediatamente
	updateBeam()
	
	-- Loop de atualização do beam
	beamUpdateLoop = true
	task.spawn(function()
		while beamUpdateLoop and myBaseESPEnabled do
			pcall(updateBeam)
			task.wait(2)
		end
	end)
	
	-- Loop de atualização do gradiente
	gradientUpdateLoop = true
	task.spawn(function()
		while gradientUpdateLoop and myBaseESPEnabled do
			local plot = findOwnedPlot()
			if plot then
				if #cachedParts == 0 then
					cachePlotParts(plot)
				end
				updateGradient()
			else
				cachedParts = {}
			end
			task.wait(UPDATE_DELAY)
		end
	end)
end

local function setupPlayerESP()
	if playerESPEnabled then
		-- Aplica ESP em todos os jogadores
		for _, plr in ipairs(Services.Players:GetPlayers()) do
			if plr ~= player then
				watchCharacter(plr)
			end
		end
		
		-- Monitora novos jogadores
		Services.Players.PlayerAdded:Connect(function(plr)
			if plr ~= player and playerESPEnabled then
				watchCharacter(plr)
			end
		end)
		
		-- Loop de transição suave de cor usando lerp
		task.spawn(function()
			local colorWhite = Color3.fromRGB(255, 255, 255)
			local colorBlack = Color3.fromRGB(0, 0, 0)
			local lerpProgress = 0
			local lerpDirection = 1
			
			while playerESPEnabled do
				lerpProgress = lerpProgress + (0.05 * lerpDirection)
				
				if lerpProgress >= 1 then
					lerpProgress = 1
					lerpDirection = -1
				elseif lerpProgress <= 0 then
					lerpProgress = 0
					lerpDirection = 1
				end
				
				local color = colorBlack:Lerp(colorWhite, lerpProgress)
				
				for plr, data in pairs(espPlayers) do
					if data.highlight and data.highlight.Parent then
						data.highlight.FillColor = color
						data.highlight.OutlineColor = color
					end
					
					if data.character then
						local billboard = data.character:FindFirstChild(PLAYER_BILLBOARD_NAME)
						if billboard then
							local label = billboard:FindFirstChild("TextLabel")
							if label then
								label.TextColor3 = color
							end
						end
					end
				end
				
				task.wait(0.03)
			end
		end)
	else
		-- Remove ESP de todos os jogadores
		for _, plr in ipairs(Services.Players:GetPlayers()) do
			if plr ~= player and plr.Character then
				removeEsp(plr, plr.Character)
			end
		end
		espPlayers = {}
	end
end

CreateToggle("ESP", "Player ESP", function(enabled)
	playerESPEnabled = enabled
	setupPlayerESP()
	if enabled then
		Notify("Player ESP ativado!")
	else
		Notify("Player ESP desativado!")
	end
end)
end

local function removeBestPetESPs()
	for model, data in pairs(bestPetESPs) do
		if data.highlight and data.highlight.Parent then
			data.highlight:Destroy()
		end
		if data.billboard and data.billboard.Parent then
			data.billboard:Destroy()
		end
	end
	bestPetESPs = {}
end

local function updateBestPetESP()
	if not bestPetESPEnabled then
		removeBestPetESPs()
		return
	end
	
	removeBestPetESPs()
	
	local cache = GetAllPetsRaw()
	if not cache or #cache == 0 then 
		warn("[ESP BEST] Nenhum pet encontrado no cache")
		return 
	end
	
	local bestPet = nil
	
	if priorityList and #priorityList > 0 then
		for _, pIndex in ipairs(priorityList) do
			for _, a in ipairs(cache) do
				if a and a.index == pIndex then
					bestPet = a
					break
				end
			end
			if bestPet then break end
		end
	end
	
	if not bestPet then
		bestPet = cache[1]
	end
	
	if not bestPet then 
		warn("[ESP BEST] Nenhum best pet encontrado")
		return 
	end
	
	print("[ESP BEST] Best pet:", bestPet.name, "Index:", bestPet.index, "Plot:", bestPet.plotName)
	
	local plot = Plots and Plots:FindFirstChild(bestPet.plotName)
	if not plot then 
		warn("[ESP BEST] Plot não encontrado:", bestPet.plotName)
		return 
	end
	
	local podiums = plot:FindFirstChild("AnimalPodiums")
	if not podiums then 
		warn("[ESP BEST] AnimalPodiums não encontrado no plot")
		return 
	end
	
	print("[ESP BEST] Procurando pet nos podiums...")
	for _, podium in ipairs(podiums:GetChildren()) do
		for _, child in ipairs(podium:GetDescendants()) do
			if child:IsA("Model") then
				local nameMatch = child.Name == bestPet.index or child.Name == bestPet.name
				if nameMatch then
					print("[ESP BEST] Pet encontrado! Model:", child.Name)
					createBestPetESP(child, bestPet)
					return
				end
			end
		end
	end
	warn("[ESP BEST] Pet não encontrado nos podiums. Procurando no plot inteiro...")
	for _, child in ipairs(plot:GetDescendants()) do
		if child:IsA("Model") and (child.Name == bestPet.index or child.Name == bestPet.name) then
			print("[ESP BEST] Pet encontrado no plot! Model:", child.Name)
			createBestPetESP(child, bestPet)
			return
		end
	end
	warn("[ESP BEST] Pet não encontrado em lugar nenhum!")
end

	local function makeVP(p, char)
		local v = Instance.new("ViewportFrame", p)
		v.Size = UDim2.fromOffset(50, 50)
		v.BackgroundColor3 = Color3.fromRGB(15, 15, 15)
		v.BorderSizePixel = 0
		v.ZIndex = 12
		v.Ambient = Color3.new(1, 1, 1)
		v.LightColor = Color3.new(1, 1, 1)
		Instance.new("UICorner", v).CornerRadius = UDim.new(0, 6)
		local st = Instance.new("UIStroke", v)
		st.Color = Color3.fromRGB(255, 255, 255)
		st.Thickness = 1
		st.Transparency = 0.6
		
		if char and char.Parent then
			task.spawn(function()
				task.wait(0.1)
				pcall(function()
					local c = char:Clone()
					-- Remover scripts
					for _, obj in ipairs(c:GetDescendants()) do
						if obj:IsA("Script") or obj:IsA("LocalScript") or obj:IsA("ModuleScript") then
							obj:Destroy()
						end
					end
					
					-- Posicionar no centro do viewport
					if c:FindFirstChild("HumanoidRootPart") then
						c.HumanoidRootPart.CFrame = CFrame.new(0, 0, 0)
					end
					
					c.Parent = v
					
					-- Configurar câmera
					local cam = Instance.new("Camera")
					cam.Parent = v
					v.CurrentCamera = cam
					
					local hd = c:FindFirstChild("Head")
					if hd then 
						cam.CFrame = CFrame.new(hd.Position + Vector3.new(0, 0.2, 2), hd.Position + Vector3.new(0, 0.2, 0))
						cam.FieldOfView = 30
					else
						local hrp = c:FindFirstChild("HumanoidRootPart")
						if hrp then
							cam.CFrame = CFrame.new(hrp.Position + Vector3.new(0, 1, 2.5), hrp.Position + Vector3.new(0, 1, 0))
							cam.FieldOfView = 40
						end
					end
				end)
			end)
		end
		return v
	end

-- ========================================
-- ANTI RAGDOLL SYSTEM
-- ========================================
local antiRagdollEnabled = false
local antiRagdollConnection = nil
local protectedMotors = {}

local function setupAntiRagdoll()
	local char = player.Character
	if not char then return end
	
	local humanoid = char:FindFirstChild("Humanoid")
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not humanoid or not hrp then return end
	
	-- Coleta Motor6Ds
	protectedMotors = {}
	for _, descendant in pairs(char:GetDescendants()) do
		if descendant:IsA("Motor6D") then
			table.insert(protectedMotors, descendant)
		end
	end
	
	-- Desabilita estados de ragdoll
	humanoid:SetStateEnabled(Enum.HumanoidStateType.Physics, false)
	humanoid:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
	
	-- Protege o ControlModule
	pcall(function()
		local playerScripts = player:WaitForChild("PlayerScripts", 5)
		if not playerScripts then return end
		
		local playerModule = playerScripts:WaitForChild("PlayerModule", 5)
		if not playerModule then return end
		
		local controlModule = require(playerModule:WaitForChild("ControlModule", 5))
		if not controlModule then return end
		
		-- Hook na função Disable do ControlModule
		if controlModule.Disable then
			local originalDisable = controlModule.Disable
			controlModule.Disable = function(...)
				if antiRagdollEnabled then
					warn("[Anti Ragdoll] Tentativa de desativar controles bloqueada!")
					return
				end
				return originalDisable(...)
			end
		end
		
		-- Hook na função Enable do ControlModule (caso tentem inverter)
		if controlModule.Enable then
			local originalEnable = controlModule.Enable
			controlModule.Enable = function(...)
				local ok, res = pcall(originalEnable, ...)
				-- Se Anti Effects estiver ativo, forçar flags comuns de inversão para false
				if antiEffectsEnabled then
					pcall(function()
						if controlModule.InvertY ~= nil then controlModule.InvertY = false end
						if controlModule.invertY ~= nil then controlModule.invertY = false end
						if controlModule.InvertMouse ~= nil then controlModule.InvertMouse = false end
						if controlModule.Inverted ~= nil then controlModule.Inverted = false end
					end)
					-- Também tenta acessar CameraModule se existir
					pcall(function()
						local playerScripts = player:FindFirstChild("PlayerScripts")
						if playerScripts then
							local playerModule = playerScripts:FindFirstChild("PlayerModule")
							if playerModule then
								local success, cameraModule = pcall(function() return require(playerModule:FindFirstChild("CameraModule")) end)
								if success and cameraModule then
									if cameraModule.InvertY ~= nil then cameraModule.InvertY = false end
									if cameraModule.invertY ~= nil then cameraModule.invertY = false end
								end
							end
						end
					end)
				end
				return ok and res or nil
			end
		end
	end)
	
	-- Monitora atributo RagdollEndTime
	player:GetAttributeChangedSignal("RagdollEndTime"):Connect(function()
		if not antiRagdollEnabled then return end
		local endTime = player:GetAttribute("RagdollEndTime")
		if endTime then
			local startTime = tick()
			local duration = 0.05
			
			while tick() - startTime < duration do
				for _, part in pairs(char:GetDescendants()) do
					if part:IsA("BasePart") then
						part.AssemblyLinearVelocity = Vector3.zero
						part.AssemblyAngularVelocity = Vector3.zero
					end
					
					if part:IsA("BodyVelocity") or part:IsA("BodyGyro") or part:IsA("BodyPosition") or 
					   part:IsA("BodyThrust") or part:IsA("BodyAngularVelocity") or part:IsA("BodyForce") then
						part:Destroy()
					end
					
					if part:IsA("LinearVelocity") or part:IsA("AngularVelocity") or 
					   part:IsA("AlignPosition") or part:IsA("AlignOrientation") or
					   part:IsA("VectorForce") or part:IsA("Torque") then
						part:Destroy()
					end
				end
				
				for _, motor in ipairs(protectedMotors) do
					if motor and motor.Parent then
						motor.Enabled = true
					end
				end
				
				Services.RunService.Heartbeat:Wait()
			end
		end
	end)
	
	-- Bloqueia criação de constraints de ragdoll
	char.DescendantAdded:Connect(function(descendant)
		if not antiRagdollEnabled then return end
		if descendant:IsA("BallSocketConstraint") or descendant:IsA("HingeConstraint") or descendant:IsA("NoCollisionConstraint") then
			descendant:Destroy()
		end
	end)
	
	-- Loop principal
	if antiRagdollConnection then
		antiRagdollConnection:Disconnect()
	end
	
	antiRagdollConnection = Services.RunService.Heartbeat:Connect(function()
		if not antiRagdollEnabled or not char.Parent then
			if antiRagdollConnection then
				antiRagdollConnection:Disconnect()
				antiRagdollConnection = nil
			end
			return
		end
		
		for _, motor in ipairs(protectedMotors) do
			if motor and motor.Parent then
				motor.Enabled = true
			end
		end
		
		if humanoid then
			if humanoid:GetState() == Enum.HumanoidStateType.Physics then
				humanoid:ChangeState(Enum.HumanoidStateType.GettingUp)
			end
			humanoid.PlatformStand = false
			humanoid.Sit = false
		end
		
		for _, descendant in pairs(char:GetDescendants()) do
			if descendant:IsA("BallSocketConstraint") or descendant:IsA("HingeConstraint") or descendant:IsA("NoCollisionConstraint") then
				descendant:Destroy()
			end
		end
	end)
end

-- ========================================
-- TRANSPARENT BASES SYSTEM
-- ========================================
local transparentBasesEnabled = false
local DECOR_TRANSPARENCY = 0.8
local transparentConnections = {}
local transparentCache = {}

local function applyTransparency(instance)
	if not transparentBasesEnabled then return end
	
	if instance:IsA("BasePart") then
		if not transparentCache[instance] then
			transparentCache[instance] = instance.Transparency
		end
		instance.Transparency = DECOR_TRANSPARENCY
	elseif instance:IsA("Decal") or instance:IsA("Texture") then
		if not transparentCache[instance] then
			transparentCache[instance] = instance.Transparency
		end
		instance.Transparency = DECOR_TRANSPARENCY
	end
end

local function restoreTransparency(instance)
	if transparentCache[instance] then
		instance.Transparency = transparentCache[instance]
		transparentCache[instance] = nil
	end
end

local function hookDecorations(decorations)
	if not transparentBasesEnabled then return end
	
	for _, descendant in ipairs(decorations:GetDescendants()) do
		applyTransparency(descendant)
	end
	
	local conn = decorations.DescendantAdded:Connect(function(descendant)
		if transparentBasesEnabled then
			applyTransparency(descendant)
		end
	end)
	table.insert(transparentConnections, conn)
end

local function hookPlot(plot)
	if not transparentBasesEnabled then return end
	
	local decorations = plot:FindFirstChild("Decorations")
	if decorations then
		hookDecorations(decorations)
	end
	
	local conn = plot.ChildAdded:Connect(function(child)
		if child.Name == "Decorations" and transparentBasesEnabled then
			hookDecorations(child)
		end
	end)
	table.insert(transparentConnections, conn)
end

local function setupTransparentBases()
	-- Limpa conexões antigas
	for _, conn in ipairs(transparentConnections) do
		conn:Disconnect()
	end
	transparentConnections = {}
	
	local plots = Services.Workspace:FindFirstChild("Plots")
	if not plots then
		warn("[Transparent Bases] Plots não encontrado!")
		return
	end
	
	if transparentBasesEnabled then
		-- Aplica transparência em plots existentes
		for _, plot in ipairs(plots:GetChildren()) do
			hookPlot(plot)
		end
		
		-- Monitora novos plots
		local conn = plots.ChildAdded:Connect(function(plot)
			if transparentBasesEnabled then
				hookPlot(plot)
			end
		end)
		table.insert(transparentConnections, conn)
	else
		-- Restaura transparência original
		for instance, _ in pairs(transparentCache) do
			if instance and instance.Parent then
				restoreTransparency(instance)
			end
		end
		transparentCache = {}
	end
end

-- Toggle FPS Boost (Nuclear FPS Booster)
local fpsBoostEnabled = false
local fpsOriginals = {}
local fpsAnimatorConns = {}
local fpsDescendantConn = nil
local Lighting = game:GetService("Lighting")
local Terrain = workspace:FindFirstChild("Terrain")

local function safeSetFFlag(name, value)
	pcall(function() if setfflag then setfflag(name, value) end end)
end

local function nukeVisualInstance(v)
	if not v then return end
	if v:IsA("Decal") or v:IsA("Texture") then
		v.Transparency = 1
	elseif v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") or v:IsA("Fire") or v:IsA("Smoke") or v:IsA("Sparkles") then
		v.Enabled = false
	elseif v:IsA("PointLight") or v:IsA("SpotLight") or v:IsA("SurfaceLight") then
		v.Enabled = false
	elseif v:IsA("Atmosphere") or v:IsA("Clouds") then
		pcall(function() v:Destroy() end)
	end
end

local function stopAnimator(anim)
	pcall(function()
		for _, t in ipairs(anim:GetPlayingAnimationTracks()) do
			pcall(function() t:Stop(0) end)
			pcall(function() t:Destroy() end)
		end
	end)
	if anim.AnimationPlayed then
		local conn = anim.AnimationPlayed:Connect(function(track)
			pcall(function() track:Stop(0); track:Destroy() end)
		end)
		table.insert(fpsAnimatorConns, conn)
	end
end

local function enableFPSBoost()
	if fpsBoostEnabled then return end
	fpsBoostEnabled = true

	-- Store some Lighting originals
	pcall(function()
		fpsOriginals.Lighting = {
			GlobalShadows = Lighting.GlobalShadows,
			FogStart = Lighting.FogStart,
			FogEnd = Lighting.FogEnd,
			Brightness = Lighting.Brightness,
			EnvironmentDiffuseScale = Lighting.EnvironmentDiffuseScale,
			EnvironmentSpecularScale = Lighting.EnvironmentSpecularScale,
			ClockTime = Lighting.ClockTime,
			OutdoorAmbient = Lighting.OutdoorAmbient,
		}
	end)

	-- FFlags (best-effort)
	pcall(function()
		safeSetFFlag("RenderingRemoveDecals", true)
		safeSetFFlag("AnimateLuaMemoryReduction", true)
		safeSetFFlag("HumanoidParallelRemoveNoPhysics", true)
		safeSetFFlag("StreamingEnabled", false)
		safeSetFFlag("MeshPartStreamingEnabled", false)
		safeSetFFlag("LuaDeferredAnimationDestroy", true)
		safeSetFFlag("LegacyUIFix", true)
	end)

	-- Engine force
	pcall(function()
		if settings and settings().Rendering then
			fpsOriginals.Rendering = { QualityLevel = settings().Rendering.QualityLevel, MeshCacheSize = settings().Rendering.MeshCacheSize }
			pcall(function() settings().Rendering.QualityLevel = Enum.QualityLevel.Level12 end)
			pcall(function() settings().Rendering.MeshCacheSize = 0 end)
			pcall(function() settings().Rendering.ThrottleFPS = 123123123 end)
		end
	end)

	-- Lighting / postfx
	pcall(function()
		Lighting.GlobalShadows = false
		Lighting.FogStart = 0
		Lighting.FogEnd = 9e9
		Lighting.Brightness = 0
		Lighting.EnvironmentDiffuseScale = 0
		Lighting.EnvironmentSpecularScale = 0
		Lighting.ClockTime = 0
		Lighting.OutdoorAmbient = Color3.fromRGB(128,128,128)
		for _, v in ipairs(Lighting:GetChildren()) do
			if v:IsA("PostEffect") then
				pcall(function() v.Enabled = false end)
			end
		end
	end)

	-- Terrain / water
	pcall(function()
		if Terrain then
			fpsOriginals.Terrain = {
				WaterWaveSize = Terrain.WaterWaveSize,
				WaterWaveSpeed = Terrain.WaterWaveSpeed,
				WaterReflectance = Terrain.WaterReflectance,
				WaterTransparency = Terrain.WaterTransparency,
			}
			pcall(function() Terrain.WaterWaveSize = 0 end)
			pcall(function() Terrain.WaterWaveSpeed = 0 end)
			pcall(function() Terrain.WaterReflectance = 0 end)
			pcall(function() Terrain.WaterTransparency = 1 end)
			pcall(function() Terrain:Clear() end)
		end
	end)

	-- Stop existing animators and hook new ones
	for _, v in ipairs(workspace:GetDescendants()) do
		if v:IsA("Animator") then
			pcall(stopAnimator, v)
		end
		nukeVisualInstance(v)
	end

	fpsDescendantConn = workspace.DescendantAdded:Connect(function(v)
		pcall(function()
			if v:IsA("Animator") then stopAnimator(v) end
			nukeVisualInstance(v)
			if (v:IsA("LocalScript") or v:IsA("Script")) and v.Name == "Animate" then pcall(function() v:Destroy() end) end
			if v:IsA("Decal") or v:IsA("Texture") then pcall(function() v.Transparency = 1 end) end
			if v:IsA("ParticleEmitter") or v:IsA("Trail") or v:IsA("Beam") then pcall(function() v.Enabled = false end) end
		end)
	end)

	-- Aggressive removals (best-effort)
	pcall(function()
		for _, v in ipairs(workspace:GetDescendants()) do
			if v:IsA("Atmosphere") or v:IsA("Clouds") then pcall(function() v:Destroy() end) end
			if (v:IsA("LocalScript") or v:IsA("Script")) and v.Name == "Animate" then pcall(function() v:Destroy() end) end
		end
	end)

	pcall(function() setfpscap(math.huge) end)
	Notify("FPS Boost ativado — alterações agressivas aplicadas")
end

local function disableFPSBoost()
	if not fpsBoostEnabled then return end
	fpsBoostEnabled = false

	-- Disconnect animator hooks
	for _, c in ipairs(fpsAnimatorConns) do pcall(function() c:Disconnect() end) end
	fpsAnimatorConns = {}
	if fpsDescendantConn then pcall(function() fpsDescendantConn:Disconnect() end) end
	fpsDescendantConn = nil

	-- Try restore Lighting
	pcall(function()
		if fpsOriginals.Lighting then
			local o = fpsOriginals.Lighting
			Lighting.GlobalShadows = o.GlobalShadows
			Lighting.FogStart = o.FogStart
			Lighting.FogEnd = o.FogEnd
			Lighting.Brightness = o.Brightness
			Lighting.EnvironmentDiffuseScale = o.EnvironmentDiffuseScale
			Lighting.EnvironmentSpecularScale = o.EnvironmentSpecularScale
			Lighting.ClockTime = o.ClockTime
			Lighting.OutdoorAmbient = o.OutdoorAmbient
		end
		if fpsOriginals.Rendering and settings and settings().Rendering then
			pcall(function() settings().Rendering.QualityLevel = fpsOriginals.Rendering.QualityLevel end)
			pcall(function() settings().Rendering.MeshCacheSize = fpsOriginals.Rendering.MeshCacheSize end)
		end
		if fpsOriginals.Terrain and Terrain then
			Terrain.WaterWaveSize = fpsOriginals.Terrain.WaterWaveSize
			Terrain.WaterWaveSpeed = fpsOriginals.Terrain.WaterWaveSpeed
			Terrain.WaterReflectance = fpsOriginals.Terrain.WaterReflectance
			Terrain.WaterTransparency = fpsOriginals.Terrain.WaterTransparency
		end
	end)

	-- Note: destruições e algumas alterações podem ser irreversíveis
	Notify("FPS Boost desativado — algumas alterações podem não ser reversíveis")
end

	if now - LagbackSystem.lastTime < 0.05 then return end

		LagbackSystem.callCount = 0
		LagbackSystem.windowStart = now
	end

	if LagbackSystem.callCount >= 7 then
		createErrorOrb()
		return
	end
	
	for _, g in pairs(LagbackSystem.serverGhosts) do
		if g and g.Parent then g:Destroy() end
	end
	LagbackSystem.serverGhosts = {}

	sg.Name = "LagbackNotification"
	sg.ResetOnSpawn = false
	sg.Parent = player:WaitForChild("PlayerGui")

	sl.Size = UDim2.new(0, 500, 0, 30)

local function clearAllGhosts()
	for _, ghost in pairs(LagbackSystem.serverGhosts) do
		pcall(function()
			if ghost and ghost.Parent then ghost:Destroy() end
		end)
	end
	LagbackSystem.serverGhosts = {}
	clearErrorOrb()
	LagbackSystem.callCount = 0
	LagbackSystem.lastTime = 0
	
	pcall(function()
		local pg = player:FindFirstChild("PlayerGui")
		if pg then
			for _, gui in pairs(pg:GetChildren()) do
				if gui.Name == "LagbackNotification" then
					gui:Destroy()
				end
			end
		end
	end)
	
	pcall(function()
		if Workspace.CurrentCamera then
			for _, c in pairs(Workspace.CurrentCamera:GetChildren()) do
				if c.Name == "LagbackGhost" then
					c:Destroy()
				end
			end
		end
	end)
	
	pcall(function()
		for _, c in pairs(Workspace:GetDescendants()) do
			if c.Name == "LagbackGhost" then
				c:Destroy()
			end
		end
	end)
end

local function setupDesyncVisualizer()
	if desyncData.enabled then

		desyncData.part = Instance.new("Part")
		desyncData.part.Name = "DesyncVisualizer"
		desyncData.part.Shape = Enum.PartType.Ball
		desyncData.part.Size = Vector3.new(2, 2, 2)
		desyncData.part.Color = Color3.fromRGB(0, 0, 0)
		desyncData.part.Material = Enum.Material.Neon
		desyncData.part.Transparency = 0.5
		desyncData.part.Anchored = true
		desyncData.part.CanCollide = false
		desyncData.part.CastShadow = false
		desyncData.part.Parent = Workspace
		
		desyncData.attachments[1] = Instance.new("Attachment", desyncData.part)
		desyncData.attachments[2] = Instance.new("Attachment")
		
		desyncData.beam = Instance.new("Beam")
		desyncData.beam.Name = "DesyncBeam"
		desyncData.beam.Transparency = NumberSequence.new(0.5)
		desyncData.beam.Color = ColorSequence.new(Color3.new(1, 1, 1))
		desyncData.beam.Width0 = 0.05
		desyncData.beam.Width1 = 0.05
		desyncData.beam.FaceCamera = true
		desyncData.beam.Attachment0 = desyncData.attachments[2]
		desyncData.beam.Attachment1 = desyncData.attachments[1]
		desyncData.beam.Parent = desyncData.part
		
		desyncData.billboard = Instance.new("BillboardGui")
		desyncData.billboard.Name = "DesyncGui"
		desyncData.billboard.Size = UDim2.new(0, 200, 0, 50)
		desyncData.billboard.StudsOffset = Vector3.new(0, 3, 0)
		desyncData.billboard.AlwaysOnTop = true
		desyncData.billboard.Parent = desyncData.part
		
		local lbl = Instance.new("TextLabel")
		lbl.Size = UDim2.new(1, 0, 1, 0)
		lbl.BackgroundTransparency = 1
		lbl.Font = Enum.Font.GothamBlack
		lbl.TextColor3 = Color3.new(1, 1, 1)
		lbl.TextStrokeColor3 = Color3.new(0, 0, 0)
		lbl.TextStrokeTransparency = 0
		lbl.TextSize = 12
		lbl.Text = "SERVER POS\n0.0 studs"
		lbl.Parent = desyncData.billboard
		
		-- Setup inicial
		local char = player.Character
		if char then
			local hrp = char:FindFirstChild("HumanoidRootPart")
			if hrp then
				desyncData.serverPos = hrp.Position
				desyncData.lastPos = hrp.Position
				
				if desyncData.connection then
					desyncData.connection:Disconnect()
				end
				
				desyncData.connection = hrp:GetPropertyChangedSignal("CFrame"):Connect(function()
					if desyncData.enabled then
						desyncData.serverPos = hrp.Position
					end
				end)
			end
		end
		
		-- Loop de atualização
		Services.RunService.Heartbeat:Connect(function()
			if not desyncData.enabled or not desyncData.part or not desyncData.part.Parent then return end
			
			local char = player.Character
			if not char then return end
			
			local hrp = char:FindFirstChild("HumanoidRootPart")
			if not hrp then return end
			
			desyncData.attachments[2].Parent = hrp
			
			local pos = hrp.Position
			local dist = (pos - desyncData.lastPos).Magnitude
			
			if dist > 25 then
				desyncData.serverPos = pos
			end
			
			desyncData.part.Position = desyncData.serverPos + Vector3.new(0, -2.5, 0)
			desyncData.lastPos = pos
			
			local distToServer = (desyncData.serverPos - pos).Magnitude
			local lbl = desyncData.billboard:FindFirstChild("TextLabel")
			if lbl then
				lbl.Text = string.format("SERVER POS\n%.1f studs", distToServer)
			end
		end)
		
	else
		-- Remover visualizador
		if desyncData.connection then
			desyncData.connection:Disconnect()
			desyncData.connection = nil
		end
		
		if desyncData.part and desyncData.part.Parent then
			desyncData.part:Destroy()
		end
		
		desyncData.part = nil
		desyncData.beam = nil
		desyncData.billboard = nil
		desyncData.attachments = {}
	end
end

-- ========================================
-- DESYNC CONTROL SYSTEM
-- ========================================
local desyncEnabled = false
local desyncHook = nil
local desyncStealConn = nil

local function setupDesync()
	if desyncEnabled then
		-- Ativar hook permanentemente enquanto o toggle estiver ON
		if raknet and raknet.add_send_hook then
			if not desyncHook then
				desyncHook = function(packet)
					if packet.PacketId == 0x1B then
						local b = packet.AsBuffer
						buffer.writeu32(b, 1, 0xFFFFFFFF)
						buffer.writeu32(b, 5, 0xFFFFFFFF)
						buffer.writeu32(b, 9, 0xFFFFFFFF)
						packet:SetData(b)
					end
				end
				raknet.add_send_hook(desyncHook)
			end
		else
			warn("[Desync] Raknet não disponível!")
		end

		-- Teleportar 0.1 studs pra frente imediatamente ao ativar
		local char = player.Character
		if char then
			local hrp = char:FindFirstChild("HumanoidRootPart")
			if hrp then
				hrp.CFrame = hrp.CFrame + (hrp.CFrame.LookVector * 0.1)
			end

			-- Usar Flying Carpet 1 frame após ativar o desync
			task.defer(function()
				local humanoid = char:FindFirstChildOfClass("Humanoid")
				if not humanoid then return end

				local toolName = "Flying Carpet"
				local tool = player.Backpack:FindFirstChild(toolName) or char:FindFirstChild(toolName)

				if tool and tool:IsA("Tool") then
					humanoid:EquipTool(tool)
					task.wait() -- Garante que equipou
					tool:Activate()
					task.wait(0.05)
					humanoid:UnequipTools()
				else
					warn("Flying Carpet não encontrado para o Desync.")
				end
			end)
		end
	else
		-- Desativa o hook permanentemente ao desligar o toggle
		if raknet and raknet.remove_send_hook and desyncHook then
			raknet.remove_send_hook(desyncHook)
			desyncHook = nil
		end
	end
end

CreateToggle("Settings", "Enable Desync", function(enabled)
	desyncEnabled = enabled
	setupDesync()
	if enabled then
		Notify("Desync ativado!")
	else
		Notify("Desync desativado!")
	end
end)

-- ANTI EFFECTS SYSTEM
-- ========================================
local AntiEffects = {
	enabled = false,
	conn = nil
}

local function enforceAntiEffects(enabled)
	if enabled then
		-- Start enforcing FOV = 70 every frame
		if AntiEffects.conn then AntiEffects.conn:Disconnect(); AntiEffects.conn = nil end
		AntiEffects.conn = Services.RunService.Heartbeat:Connect(function()
			local cam = Services.Workspace.CurrentCamera
			if cam and cam.FieldOfView ~= 70 then
				cam.FieldOfView = 70
			end
		end)
		-- Also attempt immediate fix
		pcall(function()
			local cam = Services.Workspace.CurrentCamera
			if cam then cam.FieldOfView = 70 end
		end)
		Notify("Anti Effects ativado: FOV limitado a 70 e controles protegidos.")
	else
		if AntiEffects.conn then
			AntiEffects.conn:Disconnect()
			AntiEffects.conn = nil
		end
		Notify("Anti Effects desativado.")
	end
end

CreateToggle("Settings", "Anti Effects", function(enabled)
	AntiEffects.enabled = enabled
	enforceAntiEffects(enabled)
end)
